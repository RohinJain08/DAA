
def Place_Queen(x, r, c):

    for i in range(r):
        if x[i][c] == 'Q':
            return False

    (i, j) = (r, c)

    while i >= 0 and j >= 0:
        if x[i][j] == 'Q':
            return False
        i = i - 1
        j = j - 1

    (i, j) = (r, c)

    while i >= 0 and j < len(x):
        if x[i][j] == 'Q':
            return False
        i = i - 1
        j = j + 1

    return True
